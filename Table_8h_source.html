196 LookupID Table<T>::Insert(T item) {
197 
198  if (!deadList.empty()) {
199  size_t id = deadList.top();
200  entryList[id] = item;
201  entryValid[id] = true;

203  return LookupID(thisID, LookupLibrary::MapTableData(thisID, id));

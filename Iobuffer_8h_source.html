216 template <typename T>
217 T Dynacoe::InputBuffer::Read() {
218  T object;
219  std::vector<uint8_t> data = ReadBytes(sizeof(T));
220  if (!data.size()) return T();
221  memcpy(&object, &data[0], sizeof(object));
222  return object;
